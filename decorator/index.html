Python装饰器

装饰器

一开始接触Python的装饰器，以为他跟Java的注解很像，但实际上他们两个完全不一样，Python的装饰器实际上就是一个函数，参数是函数，返回值也是函数，本质上就是个语法糖。

import time


def timer(func):
    def wrapper():
        time_start = time.time()
        func()
        time_end = time.time()
        print(f'Time cost = {time_end - time_start:.3f}s')

    return wrapper


@timer
def say_hello():
    time.sleep(3)
    print('hello')


say_hello()

运行结果

hello
Time cost = 3.000s

以上写法等同于以下写法

import time


def timer(func):
    def wrapper():
        time_start = time.time()
        func()
        time_end = time.time()
        print(f'Time cost = {time_end - time_start:.3f}s')

    return wrapper


def say_hello():
    time.sleep(3)
    print('hello')


timer(say_hello)()

运行结果

hello
Time cost = 3.000s

带参数的装饰器

import time


def timer(msg):
    def outer_wrapper(func):
        print(msg)

        def wrapper():
            time_start = time.time()
            func()
            time_end = time.time()
            print(f'Time cost = {time_end - time_start:.3f}s')

        return wrapper()

    return outer_wrapper


@timer(msg='计时器开始工作啦')
def say_hello():
    time.sleep(3)
    print('hello')


timer(say_hello)

运行结果

计时器开始工作啦
hello
Time cost = 3.000s

以上写法等同于以下写法

import time


def timer(msg):
    def outer_wrapper(func):
        print(msg)

        def wrapper():
            time_start = time.time()
            func()
            time_end = time.time()
            print(f'Time cost = {time_end - time_start:.3f}s')

        return wrapper

    return outer_wrapper


def say_hello():
    time.sleep(3)
    print('hello')


timer('计时器开始工作啦')(say_hello)()

运行结果

计时器开始工作啦
hello
Time cost = 3.000s

可以发现装饰器的写法更加的通俗易懂，毕竟是语法糖嘛
